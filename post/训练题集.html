CF451E | Devu and Flowers

  • 题意
    • Devu 有 $n$ 个花瓶，第 $i$ 个花瓶里有 $f_i$ 朵花。他现在要选择 $s$ 朵花。
    • 你需要求出有多少种方案。两种方案不同当且仅当两种方案中至少有一个花瓶选择花的数量不同。
    • 答案对 $10^9 + 7$ 取模。
  • 分析
    • 假如没有花瓶中对花数量的限制，我们可以用隔板法分隔出来不同的方案，我们要在 $s$ 朵花中插入 $n - 1$ 块板，第一块板有 $(s + 1)$ 种选择，最后的总数是 $(s + 1) \times (s + 2) \times \cdots \times (s + n - 1) = \frac{(s + n - 1)!}{s!}$ 。
    • 但是方案数量与板的顺序无关，所以我们还要去掉这些板全排列的数量，即 $C^{n - 1}_{s + n - 1}$ 。
    • 现在加入条件，我们只要把不符合条件的方案减去即可。
    • 首先我们要从所有的花中取出 $f_i + 1$ 朵在花瓶 $i$ 中的花，这样无论如何都是不符合条件的，方案数是 $C^{n - 1}_{s + n - 1 - (f_i + 1)}$。
    • 之后就考虑使用容斥原理，因为我们之前减掉的方案有重复的，又因为有 $1 \le n \le 20$，我们可以用状态压缩枚举每种情况，又因为组合数范围很大，我们直接使用逆元求组合数。
  • 代码
#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int P = 1e9 + 7;
int k;
LL r, n[25], inv[25];
LL qpow (LL a, LL b) {
    LL res = 1;
    while (b) {
        if (b & 1) res = res * a % P;
        a = a * a % P;
        b >>= 1;
    }
    return res;
}
void get_inv (LL n, LL p) {
    inv[1] = 1;
    for (LL i = 2; i <= n; i++) {
        inv[i] = (p - p / i) * inv[p % i] % p;
    }
}
LL C (LL n, LL m) {
    if (n < m) return 0;
    n %= P, m %= P;
    LL res = 1;
    for (LL i = 1; i <= m; i++) {
        res = res * (n - i + 1) % P;
        res = res * inv[i] % P;
    }
    return res;
}
int main() {
    get_inv(20, P);
    cin.tie(0)->sync_with_stdio(0);
    cin >> k >> r;
    LL ans = C(k + r - 1, k - 1);
    for (int i = 1; i <= k; i++) {
        cin >> n[i];
    }
    for (int x = 1; x < 1 << k; x++) {
        LL t = k + r, num = 0;
        for (int i = 0; i < k; i++) {
            if (x >> i & 1) {
                t -= n[i + 1];
                num++;
            }
        }
        t -= num + 1;
        if (num % 2 == 1) ans = (ans - C(t, k - 1) + P) % P;
        else ans = (ans + C(t, k - 1)) % P;
    }
    cout << ans << '\n';
}

SP32952 | ADAFTBLL - Ada and Football

  • 题意
    • 你有一棵共 $n$ 个节点的树，一共有 $q$ 次操作，操作有两种分别为：
      1. 1 x y ：表示把第 $x$ 个节点的权值变成 $y$。
      2. 2 u v ：输出 $\sum \frac{s \times (s - 1)}{2}$，其中 $s$ 为 $u$ 到 $v$ 节点中出现的不同权值的次数。
  • 分析
    • 树上带修莫队。
    • 树上莫队的宗旨是把树变成序列，就能按普通序列处理了，这里要用到两个知识点 ：重链剖分 + 欧拉序。
    • 对于 $a - b$ 的路径，分两种情况讨论。
      • $a$ 和 $b$ 在一条链上，途中的节点都只在 $a$ 和 $b$ 第一次出现的 dfs 序之间。
      • $a$ 和 $b$ 不在一条链上，那途中节点则为 $a$ 最后一次出现的地方到 $b$ 第一次出现的地方之间只出现一次的节点，$a$ 最后一次出现因为把 $a$ 的子树遍历完了， 之所以只出现一次和上面一样，不过最后还要加上它们的 $lca$，因为 a 和 b 都在 $lca(a, b)$ 的子树里，$lca$ 没有遍历到。
    • 注意，需要开一个数组记录节点的访问次数，如果访问一次就 del，没有就 add，然后取反即可。
  • 代码
#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int N = 100010;
int h[N], e[N << 1], ne[N << 1], idx;
void add(int a, int b) {
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

inline int read() {
    int x = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-')
            f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    return x * f;
}
inline void write(LL x) {
    if (x < 0) {
        x = -x;
        putchar('-');
    }
    if (x > 9) {
        write(x / 10);
    }
    putchar (x % 10 + '0');
}


int in[N], out[N], dfn, tops;
int fa[N], dep[N], sz[N], hson[N], top[N], dfsn[N], madfsn[N], rk[N << 2], id;
void dfs1(int u, int d = 1) {
    dep[u] = d, sz[u] = 1;
    in[u] = ++tops;
    rk[tops] = u;
    for (int i = h[u]; ~i; i = ne[i]) {
        int j = e[i];
        if (j == fa[u]) continue;
        fa[j] = u;
        dfs1(j, d + 1);
        sz[u] += sz[j];
        if (sz[j] > sz[hson[u]]) hson[u] = j;
    }
    out[u] = ++tops;
    rk[tops] = u;
}
void dfs2(int p) {
    madfsn[p] = dfsn[p] = ++dfn;
    if (hson[p] != 0) {
        top[hson[p]] = top[p];
        dfs2(hson[p]);
        madfsn[p] = max(madfsn[p], madfsn[hson[p]]);
    }
    for (int i = h[p]; ~i; i = ne[i]) {
        int j = e[i];
        if (j == fa[p] || j == hson[p]) continue;
        if (!top[j]) {
            top[j] = j;
            dfs2(j);
            madfsn[p] = max(madfsn[p], madfsn[j]);
        }
    }
}
int lca(int a, int b) {
    while (top[a] != top[b]) {
        if (dep[top[a]] < dep[top[b]]) swap(a, b);
        a = fa[top[a]];
    }
    return dep[a] < dep[b] ? a : b;
}

int cnt[N], vis[N], a[N];
LL ans[N], tot;
int sq, ccnt, qcnt;
struct Query {
    int l, r, id, lca, time;
    bool operator < (const Query &o) const {
        if (l / sq == o.l / sq) {
            if (r / sq == o.r / sq) {
                return time < o.time;
            } else {
                return r < o.r;
            }
        } else {
            return l < o.l;
        }
    }
}q[N];
struct Modify {
    int p, del;
}c[N];
inline void del(int x) {
    tot -= 1ll * cnt[x] * (cnt[x] - 1) / 2;
    cnt[x]--;
    tot += 1ll * cnt[x] * (cnt[x] - 1) / 2;
}
inline void add(int x) {
    tot -= 1ll * cnt[x] * (cnt[x] - 1) / 2;
    cnt[x]++;
    tot += 1ll * cnt[x] * (cnt[x] - 1) / 2;
}
inline void modify(int x) {
    if (vis[x] == 0) {
        add(a[x]);
    } else {
        del(a[x]);
    }
    vis[x] ^= 1;
}
inline void update(int t) {
    if (vis[c[t].p] == 1) {
        del(a[c[t].p]);
        add(c[t].del);
    }
    swap(a[c[t].p], c[t].del);
}

int main() {
    memset(h, -1, sizeof(h));
    int n = read(), m = read();
    for (int i = 1; i <= n; i++) {
        a[i] = read();
    }
    for(int i = 1; i < n; i++) {
        int a = read(), b = read();
        a++, b++;
        add(a, b), add(b, a);
    }

    top[1] = 1;
    dfs1(1);
    dfs2(1);

    for (int i = 1; i <= m; i++) {
        int op = read(), a = read(), b = read();
        if (op == 1) {
            a++;
            c[++ccnt] = {a, b};
        } else {
            a++, b++;
            if (in[a] > in[b]) swap(a, b);
            int l = lca(a, b);
            if (l == a) {
                q[++qcnt] = {in[a], in[b], qcnt, 0, ccnt};
            } else {
                q[++qcnt] = {out[a], in[b], qcnt, l, ccnt};
            }
        }
    }

    sq = pow(2 * n, 0.666666);
    sort(q + 1, q + qcnt + 1);
    int l = 1, r = 0, t = 0;
    for (int i = 1; i <= qcnt; i++) {
        int ql = q[i].l, qr = q[i].r, qt = q[i].time, qlca = q[i].lca;
        while (l < ql) modify(rk[l++]);
        while (l > ql) modify(rk[--l]);
        while (r < qr) modify(rk[++r]);
        while (r > qr) modify(rk[r--]);
        while (t > qt) update(t--);
        while (t < qt) update(++t);
        if (qlca != 0) modify(qlca);
        ans[q[i].id] = tot;
        if (qlca != 0) modify(qlca);
    }
    for (int i = 1; i <= qcnt; i++) {
        write(ans[i]);
        putchar('\n');
    }
}
